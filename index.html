🚀 Quick Start 
// Initialize SDK
const Bradicoin = require('bradicoin-sdk');

const client = new Bradicoin({
    apiKey: 'your_api_key',
    apiSecret: 'your_api_secret'
});

// Create payment
const payment = await client.createPayment({
    amount: 100,
    toAddress: 'BRD-XXXX...',
    callbackUrl: 'https://yoursite.com/webhook'
});
